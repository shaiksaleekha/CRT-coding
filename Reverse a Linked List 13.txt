class Node:
    def __init__(self,val):
        self.val=val
        self.left=None
        self.right=None
class Doubly:
    def __init__(self):
        self.head=self.tail=None
    def create(self,x):
        n=Node(x)
        if self.head==None:
            self.head=self.tail=n
        else:
            self.tail.right=n
            n.left=self.tail
            self.tail=n
    def forward(self):
        if self.head==None:
            print("No value has been inserted")
            return
        print("Original Order:")
        temp=self.head
        while temp!=None:
            print(temp.val,end=" ")
            temp=temp.right
        print()
    def backward(self):
        if self.head==None:
            return
        print("Reverse Order:")
        temp=self.tail
        while temp!=None:
            print(temp.val,end=" ")
            temp=temp.left
d=Doubly()
while True:
    x=int(input())
    if x>0:
        d.create(x)
    else:
        break
d.forward()
d.backward()
